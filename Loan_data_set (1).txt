DROP TABLE W2SB_LOAN_DETAILS;
DROP TABLE W2SB_APPLICATION_DETAILS;
DROP TABLE W2SB_BANK_DETAILS;
DROP TABLE W2SB_ADDRESSES;
DROP TABLE W2SB_CUSTOMER_DETAILS;

CREATE TABLE W2SB_ADDRESSES
(
 ADDRESS_ID NUMBER(30) NOT NULL PRIMARY KEY,
 LINE1 VARCHAR2(200) NOT NULL,
 LINE2 VARCHAR2(200) NOT NULL,
 STATE1 VARCHAR2(100) NOT NULL,
 COUNTRY VARCHAR2(100) NOT NULL
);

CREATE TABLE W2SB_BANK_DETAILS 
( 
  BANK_ID NUMBER(30) NOT NULL PRIMARY KEY,
  BANK_NAME VARCHAR2(100) NOT NULL,
  AC_NO NUMBER(30) NOT NULL,
  AC_TYPE VARCHAR2(100) NOT NULL,
  BRANCH_NAME VARCHAR2(100) NOT NULL,
  CURRENT_BALANCE NUMBER(30) NOT NULL
);

CREATE TABLE W2SB_APPLICATION_DETAILS 
(
 APPLICATION_ID NUMBER(30) NOT NULL PRIMARY KEY,
 CATEGORY VARCHAR2(100) NOT NULL,
 LOAN_AMT NUMBER(30) NOT NULL,
 OCCUPATION VARCHAR2(100) NOT NULL,
 SALARY NUMBER(30) NOT NULL,
 EXISTING_LOAN VARCHAR2(100) NOT NULL,
 REASON VARCHAR2(200) NOT NULL,
 OTHER_INCOME NUMBER(30) NOT NULL
);


CREATE TABLE W2SB_LOAN_DETAILS 
(
 LOAN_ID NUMBER(30) NOT NULL PRIMARY KEY,
 LOAN_AMT_APPROVED NUMBER(30) NOT NULL,
 LOAN_INTEREST NUMBER(30) NOT NULL,
 EMI NUMBER(30) NOT NULL,
 NO_INSTALLMENT NUMBER(30) NOT NULL
);


CREATE TABLE W2SB_CUSTOMER_DETAILS 
(
  CUSTOMER_ID NUMBER(30) NOT NULL PRIMARY KEY,
  ADDRESS_ID NUMBER(30) NOT NULL,
  BANK_ID NUMBER(30) NOT NULL,
  APPLICATION_ID NUMBER(30) NOT NULL,
  LOAN_ID NUMBER(30) NOT NULL,
  NAME VARCHAR2(200) NOT NULL,
  DOB DATE NOT NULL,
  GENDER VARCHAR2(100) NOT NULL,
  PH_NO NUMBER(20) NOT NULL,
  EMAIL VARCHAR2(100) NOT NULL,
  TXN_ID VARCHAR2(100) NOT NULL,
  LOAN_STATUS VARCHAR2(100) NOT NULL
);

ALTER TABLE W2SB_CUSTOMER_DETAILS ADD FOREIGN KEY ("ADDRESS_ID") REFERENCES W2SB_ADDRESSES("ADDRESS_ID") ENABLE;
ALTER TABLE W2SB_CUSTOMER_DETAILS ADD FOREIGN KEY ("BANK_ID") REFERENCES W2SB_BANK_DETAILS("BANK_ID") ENABLE;
ALTER TABLE W2SB_CUSTOMER_DETAILS ADD FOREIGN KEY ("APPLICATION_ID") REFERENCES W2SB_APPLICATION_DETAILS("APPLICATION_ID") ENABLE;
ALTER TABLE W2SB_CUSTOMER_DETAILS ADD FOREIGN KEY ("LOAN_ID") REFERENCES W2SB_LOAN_DETAILS("LOAN_ID") ENABLE;
